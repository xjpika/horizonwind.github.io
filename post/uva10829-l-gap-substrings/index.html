题意

多组询问，每一次给定一个串sss和一个整数ggg，求sss中有多少个形如UVU\text{UVU}UVU的串，满足lenV=g\text{len}_V=glenV​=g。

和优秀的拆分一样，我们用分块枚举UVU\text{UVU}UVU中U\text{U}U的长度len\text{len}len。

这样保证每一个U\text{U}U都跨越一个块端点。

然后不一样的是，我们枚举块左端点为lll，枚举另一个右端点rrr为l+len+gl+\text{len}+gl+len+g。

为什么呢？看看图，感性理解一下：

红棕色的表示块端点，绿线先表示V\text{V}V串长度。

然后我们根据套路求LCP,LCS\text{LCP},\text{LCS}LCP,LCS

分别是红线和蓝线。

然后我们看看有没有满足条件的UVU\text{UVU}UVU串？

看看下图

红棕框表示U\text{U}U，绿框表示V\text{V}V。

把整个UVU\text{UVU}UVU的串移动下，算一算，发现其左端点的取值范围是[l−LCS+1,l−len+LCP][l-\text{LCS}+1,l-\text{len}+\text{LCP}][l−LCS+1,l−len+LCP]。不懂的可以看看上图的两个串，其两端的U\text{U}U是相同的。

那么答案加上LCP+LCS−len−1\text{LCP}+\text{LCS}-\text{len}-1LCP+LCS−len−1即可。这道题做完了？

小细节

可以正反做两次后缀数组，分别求LCS\text{LCS}LCS和LCP\text{LCP}LCP。

还有，为了不算重复，我们要规定一个块内统计的U\text{U}U串左端点一定要在块内！所以我们求出的LCS,LCP\text{LCS},\text{LCP}LCS,LCP要对len\text{len}len取min⁡\minmin...

#include<bits/stdc++.h>
#define ll long long
#define ljc 998244353

inline char gc(){
    static char now[1<<16],*S,*T;
    if (T==S){T=(S=now)+fread(now,1,1<<16,stdin);if (T==S) return EOF;}

}
#endif
inline ll read(){
    register ll x=0,f=1;char ch=gc();
    while (!isdigit(ch)){if(ch=='-')f=-1;ch=gc();}
    while (isdigit(ch)){x=(x<<3)+(x<<1)+ch-'0';ch=gc();}
    return (f==1)?x:-x;
}
char s[200001];
int n,m,T,g;
ll ans; 
struct SA{
    int a[100021],c[100021],F[300001][22],sa[100021],ra[100021],hei[100021],pos[100021];
    inline void clr(int n){
        memset(sa,0,sizeof(sa));
    	memset(ra,0,sizeof(ra));
    	memset(pos,0,sizeof(pos));
    	memset(a,0,sizeof(a));
    	memset(hei,0,sizeof(hei));
    }
    inline void qsort(int n,int m){
        for (register int i=0;i<=m;i++) c[i]=0;
        for (register int i=1;i<=n;i++) c[ra[i]]++;
        for (register int i=1;i<=m;i++) c[i]+=c[i-1];
        for (register int i=n;i;i--) sa[c[ra[pos[i]]]--]=pos[i];
    }
    inline void GetSA(int n){
        for (register int i=1;i<=n;i++) ra[i]=a[i],pos[i]=i;
        int m=100;
        qsort(n,m);
        for (register int w=1,p=0;p<n;m=p,w<<=1){
            p=0;
            for (register int i=1;i<=w;i++) pos[++p]=n-w+i;
            for (register int i=1;i<=n;i++){
                if (sa[i]>w) pos[++p]=sa[i]-w;
            }
            qsort(n,m);
            swap(pos,ra);
            ra[sa[1]]=p=1;
            for (register int i=2;i<=n;i++){
                if (pos[sa[i-1]]==pos[sa[i]]&&pos[sa[i-1]+w]==pos[sa[i]+w]){
                    ra[sa[i]]=p;
                }else ra[sa[i]]=++p;
            }
        }
    }
    inline void GetHeight(int n){
        register int ans=0;
        for (register int i=1;i<=n;i++){
            if (ans) ans--;
            register int j=sa[ra[i]-1];
            while (a[j+ans]==a[i+ans]) ans++;
            hei[ra[i]]=ans;
        }   
    }
    inline void GetST(int n){
        for (int i=1;i<=n;i++) F[i][0]=hei[i];
        for (int j=1;j<=20;j++){
            for (int i=1;i+(1<<j)<=n+1;i++){
                F[i][j]=min(F[i][j-1],F[i+(1<<(j-1))][j-1]);
            }
        }
    }
    inline int query(int l,int r){
        if (l>r) swap(l,r); l++;
        int k=log2(r-l+1);
        return min(F[l][k],F[r-(1<<k)+1][k]);
    }
}lcs,lcp;
signed main(){
	cin>>T;
    for (int ce=1;ce<=T;ce++){
        scanf("%d%s",&g,s+1);n=strlen(s+1);lcs.clr(n);lcp.clr(n);
        for (int i=1;i<=n;i++) lcs.a[n-i+1]=lcp.a[i]=s[i]-'a'+1;
        lcs.GetSA(n);lcp.GetSA(n);ans=0;
        lcs.GetHeight(n);lcp.GetHeight(n);lcs.GetST(n);lcp.GetST(n);
        for (int len=1;len<=n;len++){
            for (int i=1;i<=n;i+=len){
                int l=i,r=i+len+g;//lcp
                if (r>n) break;
                int R=n-l+1,L=n-r+1;//lcs
                int LCS=min(len,lcs.query(lcs.ra[L],lcs.ra[R]));
                int LCP=min(len,lcp.query(lcp.ra[l],lcp.ra[r]));
                int cnt=LCS+LCP-len-(LCS!=0&&LCP!=0)+1;
                if (cnt>0) ans+=cnt;
            }
        }
        printf("Case %d: %lld\n",ce,ans);
    }
    return 0;
}
