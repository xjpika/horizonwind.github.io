洛谷6198 单调栈

2020-06-22

题解全是神仙分治啊，这里给一个不用动脑子的解法。

首先考虑分析上下界，先考虑第一个数，显然===序列中111的个数。然后删去这些111分成子问题。这个问题有后效性，我们发现−1-1−1越大后效性越小（很不严谨），所以Si=Si−1+1S_i=S_{i-1}+1Si​=Si−1​+1就好了。

这不是重点，关键看我们怎么构造。

想到第一步却想不到分治真的没谁了

我们发现可以用SSS数组模拟单调栈。这样模拟的过程中我们可以得到O(n)O(n)O(n)个元素的大小限制。现在相当于给你一堆限制构造最小的字典序。

显然这个问题就可以用 拓扑排序+堆 解决了，时间复杂度O(nlog⁡2n)O(n\log_2n)O(nlog2​n)。

早知道题这么水就打洛谷月赛了，送了300分

#include<bits/stdc++.h>
#define ll long long
#define ljc 998244353
using namespace std;
int n,m,top,head[1010101],sta[1010101],indo[1010101],a[1010101];
vector<int> e[1010101];
inline void made(int from,int to){
	e[from].push_back(to);indo[to]++;
}
signed main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if (a[i]==-1) a[i]=a[i-1]+1;
	}
	for (int i=1;i<=n;i++){
		for (;top>=a[i];) made(i,sta[top--]);
		sta[++top]=i;
	}
	priority_queue<int> dl;
	while (!dl.empty()) dl.pop();
	for (int i=1;i<=n;i++){
		sta[i]=0;
		if (indo[i]==0) dl.push(-i);
	}
	top=0;
	while (!dl.empty()){
		int u=-dl.top();dl.pop();
		sta[u]=++top;
		for (auto v:e[u]){
			if (--indo[v]==0) dl.push(-v);
		}
	}
	for (int i=1;i<=n;i++) printf("%d ",sta[i]);
    return 0;
}
