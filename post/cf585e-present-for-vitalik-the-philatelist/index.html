给一个只用 222 次 FMT 的题解...看不太懂其他人怎么做的...

首先枚举数xxx，这个大家都想得到。

设fif_ifi​表示所有非空集合gcd⁡=i\gcd=igcd=i的方案数。

cnti\text{cnt}_icnti​表示数组iii的个数。

所以枚举gcd⁡\gcdgcd，答案就是

∑i=1MAXcnti∑gcd⁡(d,i)=1fd\sum_{i=1}^{\text{MAX}}\text{cnt}_i\sum_{\gcd(d,i)=1}f_d i=1∑MAX​cnti​gcd(d,i)=1∑​fd​

经验丰富选手看一眼就知道这个不太能做，也没有什么奇怪的数论性质，于是来一次基本容斥：

=∑i=1MAXcnti(∑d=1fd−∑gcd⁡(d,i)=1fd)=\sum_{i=1}^{\text{MAX}}\text{cnt}_i (\sum_{d=1}f_d-\sum_{\gcd(d,i)\not=1}f_d)=i=1∑MAX​cnti​(d=1∑​fd​−gcd(d,i)​=1∑​fd​)

=∑i=1MAXcnti(2n−1−∑gcd⁡(d,i)=1fd)=\sum_{i=1}^{\text{MAX}}\text{cnt}_i (2^n-1-\sum_{\gcd(d,i)\not=1}f_d)=i=1∑MAX​cnti​(2n−1−gcd(d,i)​=1∑​fd​)

值域很大很大，考虑如何O(1)O(1)O(1)计算

∑gcd⁡(d,i)=1fd\sum_{\gcd(d,i)\not=1}f_d gcd(d,i)​=1∑​fd​

fff怎么计算？容斥或者莫比乌斯反演。我一开始是直接用容斥的（虽然本质一样），这里写个莫比乌斯反演的，非常方便：

设

Fi=∑i∣dfdF_i=\sum_{i|d}f_d Fi​=i∣d∑​fd​

表示所有非空集合i∣gcd⁡i|\gcdi∣gcd的方案数。显然

=2∑i∣dcntd=2^{\sum_{i|d}\text{cnt}_d} =2∑i∣d​cntd​

然后用反演定理得

fi=∑i∣dμ(di)2∑i∣dcntdf_i=\sum_{i|d}\mu(\frac di)2^{\sum_{i|d}\text{cnt}_d} fi​=i∣d∑​μ(id​)2∑i∣d​cntd​

直接计算是O(nln⁡n)O(n\ln n)O(nlnn)的，很不优秀，先用求 FMT(and) 点值的方法算出小∑\sum∑，再求222的幂得到FFF，最后用求 IFMT(and) 点值的方法求出fff，O(nlog⁡log⁡n)O(n\log\log n)O(nloglogn)。

事实上不需要求fff。我们继续看：

∑gcd⁡(d,i)=1fd\sum_{\gcd(d,i)\not=1}f_d gcd(d,i)​=1∑​fd​

直接做也有些棘手，继续考虑它的数论意义：

设

X=∏pjX=\prod p_j X=∏pj​

即其质因子的乘积。

那么

∑gcd⁡(d,i)=1fd=∑gcd⁡(d,X)=1fX\sum_{\gcd(d,i)\not=1}f_d=\sum_{\gcd(d,X)\not=1}f_X gcd(d,i)​=1∑​fd​=gcd(d,X)​=1∑​fX​

就是指所有与XXX不互质的数的个数。我们知道这个东西等价于所有质因子的倍数的交集的fff之和，而这个容斥系数我们又很熟悉，就是−μ-\mu−μ。那么：

∑gcd⁡(d,i)=1fd=∑d=1,d∣X−μ(d)∑d∣ifi\sum_{\gcd(d,i)\not=1}f_d=\sum_{d\not=1,d|X}-\mu(d)\sum_{d|i}f_i gcd(d,i)​=1∑​fd​=d​=1,d∣X∑​−μ(d)d∣i∑​fi​

=∑d=1,d∣X−μ(d)Fd=\sum_{d\not=1,d|X}-\mu(d)F_d =d​=1,d∣X∑​−μ(d)Fd​

直接计算是O(nln⁡n)O(n\ln n)O(nlnn)的，很不优秀，用求 FMT(or) 点值的方法（狄利克雷前缀和）可以做到O(nlog⁡log⁡n)O(n\log\log n)O(nloglogn)。其实可以把第一次求出的FFF直接拿来乘−μ-\mu−μ再做一次就好了。所以不需要求fff。

一次 FMTand，一次 FMTor ，真是一道烧脑题，不过思路还挺顺的。时间复杂度O(nlog⁡log⁡n)O(n\log\log n)O(nloglogn)，但是跑得很慢很慢？

#include<bits/stdc++.h>
#define ll long long
#define ljc 1000000007
using namespace std;
#define gc getchar
inline ll read(){
    register ll x=0,f=1;char ch=gc();
    while (!isdigit(ch)){if(ch=='-')f=-1;ch=gc();}
    while (isdigit(ch)){x=(x<<3)+(x<<1)+ch-'0';ch=gc();}
    return (f==1)?x:-x;
}
int cnt[10101010],X[10101010];
bool vis[10101010];
ll F[10101010],pw2[1101010];
int tot,p[1010101],a[1010101],lim,mu[10101010];
inline void seive(int n){
	for (int i=2;i<=n;i++){
		if (!vis[i]) p[++tot]=i,X[i]=i,mu[i]=-1;
		for (int j=1;j<=tot;j++){
			if (1ll*p[j]*i>n) break;
			vis[p[j]*i]=1,mu[p[j]*i]=-mu[i];
			if (i%p[j]==0){
				mu[i*p[j]]=0;X[i*p[j]]=X[i];break;
			}
			else X[i*p[j]]=X[i]*p[j];
		}
	}
}
inline void FMTand(ll *F,int lim){
	for (int i=1;i<=tot;i++){
    	for (int j=lim/p[i];j;j--){
    		F[j]=(F[j]+F[j*p[i]]%ljc)%ljc;
		}
	}
}
inline void FMTor(ll *F,int lim){
	for (int i=1;i<=tot;i++){
    	for (int j=1;j*p[i]<=lim;j++){
    		F[j*p[i]]=(F[j*p[i]]+F[j])%ljc;
		}
	}
}
signed main(){
	ll n=read();
	pw2[0]=1;
	for (int i=1;i<=n;i++) pw2[i]=1ll*pw2[i-1]*2%ljc;
	for (int i=1;i<=n;i++){
		a[i]=read();cnt[a[i]]++;
		lim=max(lim,a[i]);F[a[i]]=cnt[a[i]]; 
	}
	seive(lim);
	FMTand(F,lim);
	ll ONE=pw2[F[1]]-1,_=0;
	for (int i=2;i<=lim;i++){
		ONE=(ljc+ONE+mu[i]*(pw2[F[i]]-1%ljc)%ljc)%ljc;
	}
	for (int i=1;i<=lim;i++){
		F[i]=(ljc+(pw2[F[i]]-1)*(ljc-mu[i])%ljc)%ljc;
	}
	FMTor(F,lim);
	for (int i=1;i<=lim;i++){
		_=(_+1ll*cnt[i]*(pw2[n]-1-ONE-F[X[i]]+2*ljc)%ljc)%ljc;
	}
	cout<<_;
	return 0;
}
