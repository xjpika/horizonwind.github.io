{x≡B(modA)x≡bi(modai)\left\{\begin{aligned}x\equiv B \pmod A\\x\equiv b_i\pmod {a_i}\\\end{aligned}\right. {x≡B(modA)x≡bi​(modai​)​

gcd⁡(A,ai)∣(b−B)\gcd(A,a_i)|(b-B) gcd(A,ai​)∣(b−B)

算法？就是方程两两合并。最后的余项就是答案。

void exgcd(ll a,ll b,ll &x,ll &y){
    if (!b) return (void)(x=1,y=0);
    exgcd(b,a%b,x,y);
    ll tmp=x;x=y;y=tmp-a/b*y;
}
ll gcd(ll a,ll b){
    if (b==0) return a;
    return gcd(b,a%b); 
}
inline ll lcm(ll a,ll b){
    return a/gcd(a,b)*b;
}
inline ll mabs(ll x){
    return (x>0?x:-x);
}
inline ll fast_mul(ll a,ll b,ll p){
	ll t=0;a%=p;
	while (b){
		if (b&1LL) t=(t+a)%p;
		b>>=1LL;a=(a+a)%p;
	}
	return t;
}
ll a[1000001],A,B,b[1000001],n;
signed main(){
    n=read();
    //x=b (mod a)
    for (int i=1;i<=n;i++){
        a[i]=read();b[i]=read();
    }
    A=a[1],B=b[1];
    ll x,y;
    for (int i=2;i<=n;i++){
        ll G=gcd(A,a[i]);
        if (mabs(b[i]-B)%G){
            printf("No solution!\n");
            return 0;
        }
        exgcd(A,a[i],x,y);
        x=fast_mul(x%(a[i]/G)+(a[i]/G),((b[i]-B)/G)%(a[i]/G)+a[i]/G,a[i]/G);
        ll mod=lcm(A,a[i]);B=(fast_mul(x,A,mod)+B)%mod;A=mod;
    }
    cout<<(long long)B;
    return 0;
}
